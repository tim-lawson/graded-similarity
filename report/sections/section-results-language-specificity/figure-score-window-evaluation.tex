\newcommand{\plotscorewindow}[2]{
  \foreach \modelname in #2
    {
      \addplot+[
        discard if not={model_name}{\modelname},
        line width=0.8pt,
        mark=none,
        sharp plot
      ] table [
          col sep=comma,
          x=window,
          y=score
        ]{#1};
      \addlegendentryexpanded{\expandafter\texttt\expandafter{\modelname}}
    }
}
