\newcommand{\plotscoretime}[2]{
  \addplot+[
    only marks,
    mark size = 0.8pt,
  ] table [
      col sep=comma,
      x = score,
      y = time,
    ]{#1};
  \addlegendentryexpanded{\expandafter\texttt\expandafter{#2}}
}
